        function selectPlatform(platform) {

            selectedPlatform = platform;

            

            // Reset all platform buttons

            document.querySelectorAll('.platform-btn').forEach(btn => {

                btn.classList.remove('active', 'border-blue-500', 'bg-blue-50');

                btn.classList.add('border-gray-200');

                btn.querySelector('svg').classList.remove('text-blue-600');

                btn.querySelector('svg').classList.add('text-gray-600');

            });

            

            // Activate selected platform

            const selectedBtn = document.getElementById(platform + '-btn');

            selectedBtn.classList.add('active', 'border-blue-500', 'bg-blue-50');

            selectedBtn.classList.remove('border-gray-200');

            selectedBtn.querySelector('svg').classList.add('text-blue-600');

            selectedBtn.querySelector('svg').classList.remove('text-gray-600');

            

            // Show/hide Instagram categories

            const categoriesDiv = document.getElementById('instagram-categories');

            if (platform === 'instagram') {

                categoriesDiv.classList.remove('hidden');

            } else {

                categoriesDiv.classList.add('hidden');

            }

            

            // Update generate button style

            const generateBtn = document.getElementById('generate-btn');

            generateBtn.classList.remove('instagram-gradient', 'tiktok-gradient', 'snapchat-gradient');

            generateBtn.classList.add(platform + '-gradient');

        }


        function selectCategory(category) {

            selectedCategory = category;

            

            // Reset all category buttons

            document.querySelectorAll('#instagram-categories button').forEach(btn => {

                btn.classList.remove('border-purple-500', 'bg-purple-50');

                btn.classList.add('border-gray-200');

            });

            

            // Activate selected category

            const selectedBtn = document.getElementById(category + '-btn');

            selectedBtn.classList.add('border-purple-500', 'bg-purple-50');

            selectedBtn.classList.remove('border-gray-200');

        }
